function mousePressed() { if (optionsVisible) { if (mouseX > width / 2 - 100 && mouseX < width / 2 + 100 && mouseY > height / 2 + 50 && mouseY < height / 2 + 80) { // Accept option score += 10; optionsVisible = false; clickedMessage = "You accepted the ghost's offer."; } else if (mouseX > width / 2 - 100 && mouseX < width / 2 + 100 && mouseY > height / 2 + 90 && mouseY < height / 2 + 120) { // Decline option optionsVisible = false; clickedMessage = "You declined the ghost's request."; } return; } for (let gravestone of gravestones) { if ( mouseX > gravestone.x && mouseX < gravestone.x + gravestone.w && mouseY > gravestone.y && mouseY < gravestone.y + gravestone.h ) { clickedMessage = gravestone.message; messageX = width / 2; messageY = height / 2; optionsVisible = true; } } }
